```python
    """
    Calcule le MACD signal line correctement.
    MACD line = EMA(fast) - EMA(slow) pour chaque barre.
    Signal = EMA(signal) sur la série MACD.

    if len(closes) <= slow + signal:
        return 0.0
    macd_series = []
    for i in range(slow, len(closes)):
        ef = self.compute_ema(closes[:i + 1], fast)
        es = self.compute_ema(closes[:i + 1], slow)
        macd_series.append(ef - es)
    if len(macd_series) < signal:
        return 0.0
    return self.compute_ema(macd_series, signal)
```

```python
"""
Working Memory — Filtre le bruit du 5min.
Consensus requis sur N bougies consécutives avant de valider un signal.
Signaux expirés (> max_signal_age_seconds) ignorés du consensus.
"""
from collections import deque
from dataclasses import dataclass
from typing import Optional
import time


@dataclass
class MemoryState:
    signal: str
    confidence: float
    timestamp: float
    strate_source: str


class WorkingMemory:
    def __init__(self, window: int = 3, required_consensus: int = 2,
                 max_signal_age_seconds: int = 90):
        self.window = window
        self.required_consensus = required_consensus
        self.max_signal_age_seconds = max_signal_age_seconds
        self.buffer: deque[MemoryState] = deque(maxlen=window)
        self.last_consensus: Optional[str] = None

    def push(self, signal: str, confidence: float, source: str) -> None:
        self.buffer.append(MemoryState(
            signal=signal,
            confidence=confidence,
            timestamp=time.time(),
            strate_source=source
        ))

    def get_consensus(self) -> Optional[tuple[str, float]]:
        """
        Retourne (signal, confidence_moyenne) si consensus atteint sur signaux frais.
        Ignore les signaux plus vieux que max_signal_age_seconds.
        """
        now = time.time()
        valid = [s for s in self.buffer
                 if now - s.timestamp <= self.max_signal_age_seconds]

        if len(valid) < self.required_consensus:
            return None

        signals = [s.signal for s in valid]
        confidences = [s.confidence for s in valid]

        for direction in ["LONG", "SHORT"]:
            count = signals.count(direction)
            if count >= self.required_consensus:
                avg_conf = sum(
                    c for s, c in zip(signals, confidences) if s == direction
                ) / count
                return (direction, avg_conf)
        return None

    def clear(self) -> None:
        self.buffer.clear()
```

```python
"""
Schema SQLite ORACLE v2 — WAL mode, 3 tables.
"""
import sqlite3
import os


def init_db(db_path: str) -> sqlite3.Connection:
    """Initialise la base, crée les tables si absentes, active WAL."""
    os.makedirs(os.path.dirname(db_path), exist_ok=True)
    conn = sqlite3.connect(db_path, check_same_thread=False)
    conn.execute("PRAGMA journal_mode=WAL")
    conn.execute("PRAGMA foreign_keys=ON")
    conn.executescript("""
        CREATE TABLE IF NOT EXISTS hebbian_weights (
            strate_name TEXT PRIMARY KEY,
            weight      REAL NOT NULL DEFAULT 1.0,
            updated_at  REAL NOT NULL
        );

        CREATE TABLE IF NOT EXISTS trades (
            id              INTEGER PRIMARY KEY AUTOINCREMENT,
            symbol          TEXT NOT NULL,
            direction       TEXT NOT NULL,
            mode            TEXT NOT NULL,
            size_usdt       REAL,
            sl_pct          REAL,
            tp_pct          REAL,
            entry_price     REAL,
            exit_price      REAL,
            pnl_pct         REAL,
            source_strates  TEXT,
            opened_at       REAL NOT NULL,
            closed_at       REAL
        );

        CREATE TABLE IF NOT EXISTS signals (
            id          INTEGER PRIMARY KEY AUTOINCREMENT,
            symbol      TEXT NOT NULL,
            direction   TEXT NOT NULL,
            strength    REAL,
            votes_json  TEXT,
            ts          REAL NOT NULL
        );
    """)
    conn.commit()
    return conn
```

```python
"""
Repository ORACLE v2 — HebbianRepository + TradeJournal.
Deux classes distinctes, même fichier SQLite.
"""
import json
import sqlite3
import time
import logging
from typing import Optional

try:
    from oracle_v2.db.schema import init_db
except ImportError:
    from db.schema import init_db

logger = logging.getLogger("ORACLE.Repository")


class HebbianRepository:
    """Persist/load poids Hebbian entre sessions."""

    def __init__(self, db_path: str):
        self._conn = init_db(db_path)

    def load_weights(self, strate_names: list) -> dict:
        """
        Retourne {strate_name: weight} pour les strates connues.
        Strates inconnues (jamais sauvées) absentes du dict — le HebbianWeightManager
        utilise 1.0 par défaut.
        """
        if not strate_names:
            return {}
        placeholders = ",".join("?" for _ in strate_names)
        cursor = self._conn.execute(
            f"SELECT strate_name, weight FROM hebbian_weights WHERE strate_name IN ({placeholders})",
            strate_names
        )
        return {row[0]: row[1] for row in cursor.fetchall()}

    def save_weight(self, strate_name: str, weight: float) -> None:
        """Upsert du poids d'une strate."""
        self._conn.execute(
            """INSERT INTO hebbian_weights (strate_name, weight, updated_at)
               VALUES (?, ?, ?)
               ON CONFLICT(strate_name) DO UPDATE SET weight=excluded.weight, updated_at=excluded.updated_at""",
            (strate_name, weight, time.time())
        )
        self._conn.commit()


class TradeJournal:
    """Journal de trades persisté. Chaque trade = open + close."""

    def __init__(self, db_path: str):
        self._conn = init_db(db_path)

    def open_trade(
        self,
        symbol: str,
        direction: str,
        mode: str,
        size_usdt: float,
        sl_pct: float,
        tp_pct: float,
        entry_price: float,
        source_strates: list,
    ) -> int:
        """Enregistre l'ouverture d'un trade. Retourne l'ID SQLite."""
        cursor = self._conn.execute(
            """INSERT INTO trades
               (symbol, direction, mode, size_usdt, sl_pct, tp_pct, entry_price, source_strates, opened_at)
               VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?)""",
            (symbol, direction, mode, size_usdt, sl_pct, tp_pct, entry_price,
             json.dumps(source_strates), time.time())
        )
        self._conn.commit()
        trade_id = cursor.lastrowid
        logger.info(f"TradeJournal: OPEN #{trade_id} {direction} {symbol} {size_usdt:.0f}$")
        return trade_id

    def close_trade(self, trade_id: int, exit_price: float, pnl_pct: float) -> None:
        """Enregistre la clôture d'un trade avec PnL réel."""
        self._conn.execute(
            "UPDATE trades SET exit_price=?, pnl_pct=?, closed_at=? WHERE id=?",
            (exit_price, pnl_pct, time.time(), trade_id)
        )
        self._conn.commit()
        outcome = "WIN" if pnl_pct > 0 else "LOSS"
        logger.info(f"TradeJournal: CLOSE #{trade_id} {outcome} PnL={pnl_pct:+.2%}")

    def get_recent(self, n: int = 50) -> list:
        """Retourne les n derniers trades clôturés (plus récent en premier)."""
        cursor = self._conn.execute(
```

```python
class HebbianWeightManager:
    """Poids évolutifs par performance réelle. Persistés via HebbianRepository."""
    def __init__(self, strate_names: list, repo=None):
        self.weights = {s: 1.0 for s in strate_names}
        self.history = {s: [] for s in strate_names}
        self.decay = 0.92
        self.boost = 1.08
        self.min_weight = 0.1
        self.max_weight = 3.0
        self._repo = repo

        if repo:
            saved = repo.load_weights(strate_names)
            self.weights.update(saved)
            if saved:
                logger.info(f"Hebbian: {len(saved)} poids chargés depuis SQLite")

    def update(self, strate: str, was_profitable: bool, pnl_pct: float) -> None:
        if strate not in self.weights:
            return
        self.history[strate].append(pnl_pct)
        if was_profitable:
            self.weights[strate] = min(self.max_weight, self.weights[strate] * self.boost)
        else:
            self.weights[strate] = max(self.min_weight, self.weights[strate] * self.decay)
        if self._repo:
            self._repo.save_weight(strate, self.weights[strate])
        logger.debug(f"Hebbian update [{strate}]: {'✅' if was_profitable else '❌'} → weight={self.weights[strate]:.3f}")

    def get_weight(self, strate: str) -> float:
        return self.weights.get(strate, 1.0)
```

```python
import asyncio

import logging

import time

CACHE_DIR = os.path.join(os.path.dirname(__file__), "cache")


def _cache_path(symbol: str, timeframe: str, days: int) -> str:
    os.makedirs(CACHE_DIR, exist_ok=True)
    return os.path.join(CACHE_DIR, f"{symbol}_{timeframe}_{days}d.json")


def _is_cache_valid(path: str, max_age_hours: int = 1) -> bool:
    if not os.path.exists(path):
        return False
    age = time.time() - os.path.getmtime(path)
    return age < max_age_hours * 3600


async def fetch_historical(
    symbol: str,
    timeframe: str = "5m",
    days: int = 90,
    binance_api_key: str = "",
    binance_secret: str = "",
) -> list:
    """
    Retourne une liste de candles dict {timestamp, open, high, low, close, volume}.
    Utilise le cache JSON si < 1h. Sinon fetch via ccxt.
    """
    cache_path = _cache_path(symbol, timeframe, days)
    if _is_cache_valid(cache_path):
        with open(cache_path, "r") as f:
            data = json.load(f)
        logger.info(f"DataLoader: cache hit {symbol}/{timeframe} ({len(data)} candles)")
        return data

    try:
        import ccxt.async_support as ccxt
    except ImportError:
        logger.error("ccxt non installé — pip install ccxt")
        return []

    limit = min(1000, (days * 24 * 60) // _timeframe_minutes(timeframe))
    exchange = ccxt.binanceusdm({
        "apiKey": binance_api_key,
        "secret": binance_secret,
        "enableRateLimit": True,
    })

    try:
        raw = await exchange.fetch_ohlcv(symbol, timeframe, limit=limit)
        candles = [
            {"timestamp": c[0], "open": c[1], "high": c[2],
             "low": c[3], "close": c[4], "volume": c[5]}
            for c in raw
        ]
        with open(cache_path, "w") as f:
            json.dump(candles, f)
        logger.info(f"DataLoader: fetched {len(candles)} candles {symbol}/{timeframe}")
        return candles
    except Exception as e:
        logger.error(f"DataLoader fetch error: {e}")
        return []
    finally:
        await exchange.close()


def _timeframe_minutes(tf: str) -> int:
    units = {"m": 1, "h": 60, "d": 1440}
    try:
        return int(tf[:-1]) * units.get(tf[-1], 1)
    except (ValueError, IndexError):
        return 5
```

```python
"""
BacktestEngine — Replay OHLCV bougie par bougie, calcul PnL par strate.
Pas de lookahead bias : chaque bougie ne voit que les bougies précédentes.
"""
from dataclasses import dataclass, field
import logging

logger = logging.getLogger("ORACLE.Backtest")

MIN_CANDLES_FOR_FEATURES = 35  # minimum pour calculer EMA26 + signal


@dataclass
class BacktestResult:
    symbol: str
    trades: list = field(default_factory=list)
    metrics: dict = field(default_factory=dict)
    strate_scores: dict = field(default_factory=dict)


class BacktestEngine:
    """
    Replay bougie par bougie sur des données historiques.
    Chaque strate vote, le parliament délibère, on simule l'entrée/sortie.
    """

    def __init__(self, strates: list, parliament, sl_multiplier: float = 1.5,
                 tp_ratio: float = 2.5, max_position_pct: float = 0.10,
                 min_sl_pct: float = 0.005, max_sl_pct: float = 0.02):
        self.strates = strates
        self.parliament = parliament
        self.sl_multiplier = sl_multiplier
        self.tp_ratio = tp_ratio
        self.max_position_pct = max_position_pct
        self.min_sl_pct = min_sl_pct
        self.max_sl_pct = max_sl_pct

    def _compute_atr(self, candles: list, period: int = 14) -> float:
        if len(candles) < 2:
            return 0.0
        trs = []
        for i in range(1, len(candles)):
            prev_c = candles[i - 1]["close"]
            c = candles[i]
            tr = max(c["high"] - c["low"],
                     abs(c["high"] - prev_c),
                     abs(c["low"] - prev_c))
            trs.append(tr)
        window = trs[-period:]
        return sum(window) / len(window) if window else 0.0

    def _strate_data(self, candles: list) -> dict:
        return {
            "ohlcv": candles,
            "features": self._compute_features(candles),
            "cascade": {"dominant_trend": "SIDEWAYS", "confidence": 0.5},
            "macro": {}
        }

    def _compute_features(self, candles: list) -> dict:
        closes = [c["close"] for c in candles]
        volumes = [c["volume"] for c in candles]
        n = len(closes)

        if n < 2:
            return {"rsi": 50.0, "macd": 0.0, "macd_signal": 0.0,
                    "volume_ratio": 1.0, "bb_position": 0.0}

        # RSI simple
        deltas = [closes[i] - closes[i-1] for i in range(1, n)]
        gains = [d for d in deltas if d > 0]
        losses = [-d for d in deltas if d < 0]
        avg_gain = sum(gains[-14:]) / 14 if len(gains) >= 14 else (sum(gains) / len(gains) if gains else 0)
        avg_loss = sum(losses[-14:]) / 14 if len(losses) >= 14 else (sum(losses) / len(losses) if losses else 1e-10)
        rsi = 100 - (100 / (1 + avg_gain / avg_loss))

        vol_ma = sum(volumes[-20:]) / 20 if len(volumes) >= 20 else volumes[-1]
        volume_ratio = volumes[-1] / vol_ma if vol_ma else 1.0

        return {
            "rsi": rsi,
            "macd": 0.0,
            "macd_signal": 0.0,
            "volume_ratio": volume_ratio,
            "bb_position": 0.0,
            "price_change_pct": (closes[-1] - closes[-2]) / closes[-2] if closes[-2] else 0.0,
        }

    def run_on_candles(self, symbol: str, candles: list) -> BacktestResult:
        """
        Replay sur une liste de candles dict.
        Retourne BacktestResult avec trades et métriques.
        """
        result = BacktestResult(symbol=symbol)
        capital = 10_000.0
        position = None  # None ou dict avec entry, sl, tp, direction, trade_id_idx

        strate_wins = {s.name: 0 for s in self.strates}
        strate_total = {s.name: 0 for s in self.strates}

        for i in range(MIN_CANDLES_FOR_FEATURES, len(candles)):
            window = candles[:i]  # no lookahead
            current = candles[i]
            price = current["close"]

            # Gestion position ouverte — check SL/TP
            if position:
                if position["direction"] == "LONG":
                    if price <= position["sl"]:
                        pnl = (price - position["entry"]) / position["entry"]
                        result.trades.append({
                            "idx": i, "direction": "LONG", "entry": position["entry"],
                            "exit": price, "pnl_pct": pnl, "exit_reason": "SL",
                            "strates": position["strates"]
                        })
                        capital *= (1 + pnl * self.max_position_pct)
                        for s in position["strates"]:
                            strate_total[s] = strate_total.get(s, 0) + 1
                        position = None
                        continue
                    elif price >= position["tp"]:
                        pnl = (price - position["entry"]) / position["entry"]
                        result.trades.append({
                            "idx": i, "direction": "LONG", "entry": position["entry"],
                            "exit": price, "pnl_pct": pnl, "exit_reason": "TP",
                            "strates": position["strates"]
                        })
                        capital *= (1 + pnl * self.max_position_pct)
                        for s in position["strates"]:
                            strate_total[s] = strate_total.get(s, 0) + 1
                            if pnl > 0:
                                strate_wins[s] = strate_wins.get(s, 0) + 1
                        position = None
                        continue
                else:  # SHORT
                    if price >= position["sl"]:
                        pnl = (position["entry"] - price) / position["entry"]
                        result.trades.append({
                            "idx": i, "direction": "SHORT", "entry": position["entry"],
                            "exit": price, "pnl_pct": pnl, "exit_reason": "SL",
                            "strates": position["strates"]
                        })
                        capital *= (1 + pnl * self.max_position_pct)
                        position = None
                        continue
                    elif price <= position["tp"]:
                        pnl = (position["entry"] - price) / position["entry"]
                        result.trades.append({
                            "idx": i, "direction": "SHORT", "entry": position["entry"],
                            "exit": price, "pnl_pct": pnl, "exit_reason": "TP",
                            "strates": position["strates"]
                        })
                        capital *= (1 + pnl * self.max_position_pct)
                        for s in position["strates"]:
                            strate_total[s] = strate_total.get(s, 0) + 1
                            if pnl > 0:
                                strate_wins[s] = strate_wins.get(s, 0) + 1
                        position = None
                        continue

            # Pas de position — écouter les strates
            if position:
                continue

            data = self._strate_data(window)
            try:
                from oracle_v2.brain.parliament import Vote
            except ImportError:
                from brain.parliament import Vote

            votes = []
            for strate in self.strates:
                r = strate.safe_analyze(data)
                if r.confidence > 0.3 and r.direction != "NEUTRAL":
                    votes.append(Vote(r.strate_name, r.direction, r.confidence, r.reasoning))

            if not votes:
                continue

            decision = self.parliament.deliberate(votes)
            if decision.direction == "NEUTRAL" or decision.strength < 0.6:
                continue

            # Ouvrir une position
            atr = self._compute_atr(window)
            sl_pct = max(self.min_sl_pct, min(self.max_sl_pct,
                         (atr / price) * self.sl_multiplier if price > 0 else self.min_sl_pct))
            tp_pct = sl_pct * self.tp_ratio

            if decision.direction == "LONG":
                position = {
                    "direction": "LONG",
                    "entry": price,
                    "sl": price * (1 - sl_pct),
                    "tp": price * (1 + tp_pct),
                    "strates": [v.strate_name for v in votes]
                }
            else:
                position = {
                    "direction": "SHORT",
                    "entry": price,
                    "sl": price * (1 + sl_pct),
                    "tp": price * (1 - tp_pct),
                    "strates": [v.strate_name for v in votes]
                }

        # Calcul métriques
        trades = result.trades
        total = len(trades)
        wins = [t for t in trades if t["pnl_pct"] > 0]
        pnls = [t["pnl_pct"] for t in trades]

        # Max drawdown
        equity = 1.0
        peak = 1.0
        max_dd = 0.0
        for p in pnls:
            equity *= (1 + p * self.max_position_pct)
            if equity > peak:
                peak = equity
            dd = (peak - equity) / peak
            if dd > max_dd:
                max_dd = dd

        # Sharpe simplifié
        import statistics
        sharpe = 0.0
        if len(pnls) > 1:
            mean_r = sum(pnls) / len(pnls)
            try:
                std_r = statistics.stdev(pnls)
                sharpe = (mean_r / std_r) * (252 ** 0.5) if std_r > 0 else 0.0
            except statistics.StatisticsError:
                sharpe = 0.0

        result.metrics = {
            "total_trades": total,
            "wins": len(wins),
            "losses": total - len(wins),
            "winrate": len(wins) / total if total > 0 else 0.0,
            "max_drawdown": max_dd,
            "sharpe_ratio": sharpe,
            "final_capital": capital,
        }

        result.strate_scores = {
            s: strate_wins.get(s, 0) / strate_total.get(s, 1)
            for s in strate_total if strate_total.get(s, 0) > 0
        }

        logger.info(
            f"Backtest {symbol}: {total} trades | winrate={result.metrics['winrate']:.0%} "
            f"| max_dd={max_dd:.2%} | sharpe={sharpe:.2f}"
        )
        return result
```

```python
    def _load_model(self, path: str) -> None:
        if not path or not os.path.exists(path):
            return
        try:
            import xgboost as xgb
            self.model = xgb.Booster()
            self.model.load_model(path)
            logger.info(f"PredictiveLayer: Modèle XGBoost chargé depuis {path}")
        except Exception as e:
            logger.warning(f"PredictiveLayer: impossible de charger XGBoost ({e}) — mode heuristique")

    def train(self, trades: list) -> bool:
        """
        Entraîne XGBoost sur les trades du backtest.
        trades: list[dict] avec keys: rsi, macd, volume_ratio, bb_position,
                price_change_pct, trend_score, orderbook_imbalance, pnl_pct
        Retourne True si entraînement réussi.
        """
        try:
            import xgboost as xgb
            import random
        except ImportError:
            logger.error("xgboost non installé — pip install xgboost")
            return False

        if len(trades) < 20:
            logger.warning(f"PredictiveLayer: pas assez de trades pour entraîner ({len(trades)} < 20)")
            return False

        X = []
        y = []
        for t in trades:
            row = [
                t.get("rsi", 50.0),
                t.get("macd", 0.0),
                t.get("volume_ratio", 1.0),
                t.get("bb_position", 0.0),
                t.get("price_change_pct", 0.0),
                t.get("trend_score", 0.0),
                t.get("orderbook_imbalance", 0.0),
            ]
            X.append(row)
            y.append(1 if t.get("pnl_pct", 0) > 0 else 0)

        # Shuffle + split 80/20
        combined = list(zip(X, y))
        random.shuffle(combined)
        X, y = zip(*combined)
        split = int(len(X) * 0.8)
        X_train, X_test = list(X[:split]), list(X[split:])
        y_train, y_test = list(y[:split]), list(y[split:])

        dtrain = xgb.DMatrix(X_train, label=y_train, feature_names=self.feature_names)
        dtest = xgb.DMatrix(X_test, label=y_test, feature_names=self.feature_names)

        params = {
            "objective": "binary:logistic",
            "eval_metric": "logloss",
            "max_depth": 4,
            "eta": 0.1,
            "subsample": 0.8,
            "colsample_bytree": 0.8,
            "seed": 42,
        }

        evals_result = {}
        self.model = xgb.train(
            params, dtrain,
            num_boost_round=200,
            evals=[(dtest, "test")],
            early_stopping_rounds=20,
            evals_result=evals_result,
            verbose_eval=False
        )

        # Accuracy sur test set
        preds = self.model.predict(dtest)
        correct = sum(1 for p, label in zip(preds, y_test) if (p > 0.5) == bool(label))
        accuracy = correct / len(y_test)
        logger.info(f"PredictiveLayer: XGBoost entraîné — accuracy={accuracy:.2%} sur {len(y_test)} exemples test")

        # Sauvegarder
        os.makedirs(os.path.dirname(self.model_path), exist_ok=True)
        self.model.save_model(self.model_path)
        logger.info(f"PredictiveLayer: modèle sauvé dans {self.model_path}")
        return True

    def predict(self, features: dict) -> Prediction:
        """Prédit via XGBoost si disponible, sinon heuristique."""
        if self.model is not None:
            return self._predict_xgboost(features)
        return self.predict_heuristic(features)

    def _predict_xgboost(self, features: dict) -> Prediction:
        try:
            import xgboost as xgb
            row = [[
                features.get("rsi", 50.0),
                features.get("macd", 0.0),
                features.get("volume_ratio", 1.0),
                features.get("bb_position", 0.0),
                features.get("price_change_pct", 0.0),
                features.get("trend_score", 0.0),
                features.get("orderbook_imbalance", 0.0),
            ]]
            dmat = xgb.DMatrix(row, feature_names=self.feature_names)
            prob = float(self.model.predict(dmat)[0])

            if prob > 0.55:
                direction = "LONG"
            elif prob < 0.45:
                direction = "SHORT"
            else:
                direction = "NEUTRAL"

            return Prediction(
                symbol=features.get("symbol", "UNKNOWN"),
                direction=direction,
                probability=prob,
                confidence=abs(prob - 0.5) * 2,
                features_used=self.feature_names,
                model_version="xgboost_v1"
            )
        except Exception as e:
            logger.warning(f"XGBoost predict error: {e} — fallback heuristic")
            return self.predict_heuristic(features)
```

```python
"""
ExecutionEngine — Place les ordres, gère les positions, enregistre le PnL.
Extrait de oracle_system.py pour isolation des responsabilités.
"""
import logging
from typing import Optional

logger = logging.getLogger("ORACLE.Execution")


class ExecutionEngine:
    """
    Responsabilité unique : traduire un signal parliament en action réelle ou paper.
    Dépend de : SafetyKernel, Brainstem, TradeJournal, connectors.
    """

    def __init__(self, config, brainstem, safety_kernel, trade_journal=None):
        self.config = config
        self.brainstem = brainstem
        self.safety_kernel = safety_kernel
        self.trade_journal = trade_journal
        self.open_positions: list = []

    async def try_execute(
        self,
        symbol: str,
        direction: str,
        confidence: float,
        decision,
        feature_vector,
        binance=None,
        narrator=None,
        alert_queue=None,
    ) -> Optional[int]:
        """
        Tente d'exécuter un trade. Retourne trade_id ou None si rejeté.
        """
        try:
            from oracle_v2.brain.safety_kernel import Order
        except ImportError:
            from brain.safety_kernel import Order

        alive, reason = self.brainstem.is_alive()
        if not alive:
            logger.warning(f"Brainstem bloqué avant exécution {symbol}: {reason}")
            return None

        if binance:
            balance = await binance.fetch_balance()
            capital = balance.get("free", 0)
        else:
            capital = 1000.0

        if capital <= 0:
            return None

        atr = feature_vector.atr if feature_vector else 0
        current_price = 0.0
        if binance:
            ticker = await binance.fetch_ticker(symbol)
            current_price = ticker.last if ticker else 0.0

        sl_pct = self.safety_kernel.compute_atr_sl(atr, current_price)
        tp_pct = sl_pct * self.config.DEFAULT_TP_RATIO
        size_usdt = capital * self.config.MAX_POSITION_SIZE_PCT * confidence

        order = Order(
            symbol=symbol, direction=direction,
            size_usdt=size_usdt, leverage=1.0,
            sl_pct=sl_pct, tp_pct=tp_pct,
            source_strate="PARLIAMENT", confidence=confidence
        )

        report = self.safety_kernel.validate(order, capital)
        if not report.cleared:
            logger.warning(f"SafetyKernel rejet {symbol}: {report.reason}")
            return None

        if report.adjusted_size:
            order.size_usdt = report.adjusted_size

        source_strates = [v.strate_name for v in decision.votes] if decision.votes else []

        if self.config.MODE == "paper":
            logger.info(
                f"[PAPER] {direction} {symbol} {order.size_usdt:.0f}$ | "
                f"SL:{sl_pct:.1%} TP:{tp_pct:.1%} | Conf:{confidence:.0%}"
            )
            if narrator:
                await narrator.announce_trade(
                    symbol=symbol, direction=direction,
                    size_usdt=order.size_usdt, sl_pct=sl_pct,
                    tp_pct=tp_pct, confidence=confidence, mode="paper"
                )
            if alert_queue:
                alert_queue.alert_trade_open(
                    symbol=symbol, direction=direction,
                    size_usdt=order.size_usdt, leverage=order.leverage,
                    sl_pct=sl_pct, tp_pct=tp_pct,
                    confidence=confidence, source="PARLIAMENT"
                )
            self.safety_kernel.register_open(order)
            trade_id = None
            if self.trade_journal:
                trade_id = self.trade_journal.open_trade(
                    symbol=symbol, direction=direction, mode="paper",
                    size_usdt=order.size_usdt, sl_pct=sl_pct, tp_pct=tp_pct,
                    entry_price=current_price, source_strates=source_strates
                )
            self.open_positions.append({
                "symbol": symbol, "order": order,
                "trade_id": trade_id, "entry_price": current_price
            })
            return trade_id

        if self.config.MODE == "live" and binance:
            side = "buy" if direction == "LONG" else "sell"
            result = await binance.place_order(
                symbol=symbol, side=side,
                amount=order.size_usdt / (current_price or 1),
                order_type="market"
            )
            if result:
                self.safety_kernel.register_open(order)
                self.brainstem.register_trade(0)
                trade_id = None
                if self.trade_journal:
                    trade_id = self.trade_journal.open_trade(
                        symbol=symbol, direction=direction, mode="live",
                        size_usdt=order.size_usdt, sl_pct=sl_pct, tp_pct=tp_pct,
                        entry_price=current_price, source_strates=source_strates
                    )
                if alert_queue:
                    alert_queue.alert_trade_open(
                        symbol=symbol, direction=direction,
                        size_usdt=order.size_usdt, leverage=order.leverage,
                        sl_pct=sl_pct, tp_pct=tp_pct,
                        confidence=confidence, source="PARLIAMENT"
                    )
                self.open_positions.append({
                    "symbol": symbol, "order": order,
                    "trade_id": trade_id, "entry_price": current_price
                })
                return trade_id

        return None
```

```python
"""

Extrait de oracle_system.py._analyze_symbol().

import logging

logger = logging.getLogger("ORACLE.SignalAggregator")


class SignalAggregator:
    """
    Responsabilité unique : analyser un symbole et retourner une décision parliament.
    Dépend de : strates, parliament, feature_layer, sensory_layer, association_cortex.
    """

    def __init__(self, config, strates_dict: dict, parliament, feature_layer,
                 sensory_layer, association_cortex, predictive_layer, legacy_bridge=None):
        self.config = config
        self.strates = strates_dict  # {"AMD": AMDStrate(), ...}
        self.parliament = parliament
        self.feature_layer = feature_layer
        self.sensory = sensory_layer
        self.association = association_cortex
        self.predictive = predictive_layer
        self.legacy_bridge = legacy_bridge

    async def analyze(self, symbol: str, poly_opps: list, binance=None) -> tuple:
        """
        Analyse un symbole. Retourne (decision, feature_vector, primary_ohlcv, votes).
        Retourne (None, None, None, []) si données insuffisantes.
        """
        try:
            from oracle_v2.brain.parliament import Vote
        except ImportError:
            from brain.parliament import Vote

        asset = symbol.replace("USDT", "").replace("USD", "")

        # Fetch données
        candles_by_tf = {}
        orderbook = {}
        if binance:
            for tf in [self.config.PRIMARY_TIMEFRAME] + self.config.SECONDARY_TIMEFRAMES:
                candles_by_tf[tf] = await binance.fetch_ohlcv(symbol, tf, limit=100)
            orderbook = await binance.fetch_orderbook(symbol, limit=20)
        else:
            return None, None, None, []

        if not candles_by_tf.get(self.config.PRIMARY_TIMEFRAME):
            return None, None, None, []

        sensory_input = self.sensory.build_sensory_input(
            symbol=symbol, candles_by_tf=candles_by_tf, orderbook=orderbook
        )
        primary_ohlcv = [c for c in sensory_input.ohlcv
                         if c.timeframe == self.config.PRIMARY_TIMEFRAME]

        features_by_tf = {}
        for tf in [self.config.PRIMARY_TIMEFRAME] + self.config.SECONDARY_TIMEFRAMES:
            tf_ohlcv = [c for c in sensory_input.ohlcv if c.timeframe == tf]
            features_by_tf[tf] = self.feature_layer.extract(tf_ohlcv, symbol, tf)

        cascade = self.feature_layer.cascade_analysis(features_by_tf)
        primary_fv = features_by_tf.get(self.config.PRIMARY_TIMEFRAME)

        predict_features = {
            "symbol": symbol,
            "rsi": primary_fv.rsi if primary_fv else 50.0,
            "macd": primary_fv.macd if primary_fv else 0.0,
            "volume_ratio": primary_fv.volume_ratio if primary_fv else 1.0,
            "bb_position": primary_fv.bb_position if primary_fv else 0.0,
            "price_change_pct": primary_fv.price_change_pct if primary_fv else 0.0,
            "trend_score": 1.0 if cascade.get("dominant_trend") == "UP" else
                          -1.0 if cascade.get("dominant_trend") == "DOWN" else 0.0,
            "orderbook_imbalance": sensory_input.orderbook_imbalance
        }
        prediction = self.predictive.predict(predict_features)

        strate_data = {
            "ohlcv": [{"open": c.open, "high": c.high, "low": c.low,
                       "close": c.close, "volume": c.volume} for c in primary_ohlcv],
            "features": {
                "rsi": predict_features["rsi"],
                "macd": predict_features["macd"],
                "macd_signal": primary_fv.macd_signal if primary_fv else 0.0,
                "volume_ratio": predict_features["volume_ratio"],
                "bb_position": predict_features["bb_position"],
                "price_change_pct": predict_features["price_change_pct"],
            },
            "cascade": cascade,
            "macro": {}
        }

        votes = []
        for strate_name, strate in self.strates.items():
            result = strate.safe_analyze(strate_data)
            if result.confidence > 0.3:
                votes.append(Vote(strate_name, result.direction, result.confidence, result.reasoning))

        if prediction.confidence > 0.3:
            votes.append(Vote("PREDICTIVE", prediction.direction, prediction.confidence,
                              f"Model: {prediction.model_version} p={prediction.probability:.2f}"))

        poly_vote = self.strates.get("POLYMARKET_STRATE")
        if poly_vote:
            pv = poly_vote.generate_parliament_vote(poly_opps, asset) if hasattr(poly_vote, "generate_parliament_vote") else None
            if pv and pv.confidence > 0:
                votes.append(pv)

        if self.legacy_bridge and self.legacy_bridge.available_strates:
            try:
                legacy_votes = self.legacy_bridge.generate_votes(
                    symbol, primary_ohlcv, features=predict_features, trade_history=[]
                )
                for v in legacy_votes:
                    if v.confidence > 0.3:
                        votes.append(v)
            except Exception as e:
                logger.debug(f"Legacy bridge error {symbol}: {e}")

        if not votes:
            return None, primary_fv, primary_ohlcv, []

        decision = self.parliament.deliberate(votes)
        return decision, primary_fv, primary_ohlcv, votes
```

```python
"""
CycleManager — Boucle principale 60s + boucle BTC latency arb 30s.
Extrait de oracle_system.py.run() et _btc_latency_loop().
"""
import asyncio
import logging
import time
from datetime import datetime, timezone, timedelta

logger = logging.getLogger("ORACLE.CycleManager")
TAHITI_TZ = timezone(timedelta(hours=-10))
CYCLE_INTERVAL = 60


class CycleManager:
    """
    Responsabilité unique : orchestrer les cycles de trading.
    Dépend de : OracleSystem pour les callbacks métier.
    """

    def __init__(self, config, oracle_system):
        self.config = config
        self.system = oracle_system
        self._running = False
        self._paused = False

    async def run(self):
        self._running = True
        logger.info("CycleManager: boucle principale démarrée")

        latency_task = asyncio.create_task(self._btc_latency_loop())

        try:
            while self._running:
                cycle_start = time.time()
                now_tahiti = datetime.now(TAHITI_TZ)
                logger.info(f"Cycle {now_tahiti.strftime('%H:%M:%S')} UTC-10")

                if not self._paused:
                    await self.system._run_cycle()
                else:
                    logger.info("Système en pause — cycle ignoré")

                elapsed = time.time() - cycle_start
                await asyncio.sleep(max(0, CYCLE_INTERVAL - elapsed))

        except asyncio.CancelledError:
            logger.info("CycleManager: arrêt demandé")
        finally:
            latency_task.cancel()

    async def _btc_latency_loop(self):
        while self._running:
            if not self._paused and self.config.BTC_LATENCY_ARB_ENABLED:
                await self.system._run_btc_latency_tick()
            await asyncio.sleep(self.config.BTC_LATENCY_ARB_INTERVAL)

    def pause(self):
        self._paused = True

    def resume(self):
        self._paused = False

    def stop(self):
        self._running = False
```

```python
            sigma=self.config.BTC_SIGMA_ANNUAL,

    async def _run_cycle(self):
        alive, reason = self.brainstem.is_alive()
        if not alive:
            logger.warning(f"Brainstem BLOQUE: {reason}")
            if self.alert_queue:
                self.alert_queue.alert_brainstem(reason)
            if self.narrator:
                self.narrator.brainstem_block(reason)
            return

        poly_opps = []
        try:
            poly_opps = await self.polymarket_strate.scan()
        except Exception as e:
            logger.warning(f"Polymarket scan failed: {e}")

        for symbol in self.config.TRADING_PAIRS:
            try:
                await self._analyze_symbol(symbol, poly_opps)
            except Exception as e:
                logger.error(f"Erreur analyse {symbol}: {e}")

    async def _analyze_symbol(self, symbol: str, poly_opps: list):
        try:
            from oracle_v2.brain.parliament import Vote
        except ImportError:
            from brain.parliament import Vote

        asset = symbol.replace("USDT", "").replace("USD", "")
        candles_by_tf = {}
        orderbook = {}
        if self.binance:
            for tf in [self.config.PRIMARY_TIMEFRAME] + self.config.SECONDARY_TIMEFRAMES:
                candles_by_tf[tf] = await self.binance.fetch_ohlcv(symbol, tf, limit=100)
            orderbook = await self.binance.fetch_orderbook(symbol, limit=20)
        else:
            return

        if not candles_by_tf.get(self.config.PRIMARY_TIMEFRAME):
            return

        sensory_input = self.sensory.build_sensory_input(
            symbol=symbol, candles_by_tf=candles_by_tf, orderbook=orderbook
        )
        primary_ohlcv = [c for c in sensory_input.ohlcv
                         if c.timeframe == self.config.PRIMARY_TIMEFRAME]
        features_by_tf = {}
        for tf in [self.config.PRIMARY_TIMEFRAME] + self.config.SECONDARY_TIMEFRAMES:
            tf_ohlcv = [c for c in sensory_input.ohlcv if c.timeframe == tf]
            features_by_tf[tf] = self.feature_layer.extract(tf_ohlcv, symbol, tf)

        cascade = self.feature_layer.cascade_analysis(features_by_tf)
        primary_fv = features_by_tf.get(self.config.PRIMARY_TIMEFRAME)

        predict_features = {
            "symbol": symbol,
            "rsi": primary_fv.rsi if primary_fv else 50.0,
            "macd": primary_fv.macd if primary_fv else 0.0,
            "volume_ratio": primary_fv.volume_ratio if primary_fv else 1.0,
            "bb_position": primary_fv.bb_position if primary_fv else 0.0,
            "price_change_pct": primary_fv.price_change_pct if primary_fv else 0.0,
            "trend_score": 1.0 if cascade.get("dominant_trend") == "UP" else
                          -1.0 if cascade.get("dominant_trend") == "DOWN" else 0.0,
            "orderbook_imbalance": sensory_input.orderbook_imbalance
        }
        prediction = self.predictive.predict(predict_features)

        strate_data = {
            "ohlcv": [{"open": c.open, "high": c.high, "low": c.low,
                       "close": c.close, "volume": c.volume} for c in primary_ohlcv],
            "features": {
                "rsi": predict_features["rsi"], "macd": predict_features["macd"],
                "macd_signal": primary_fv.macd_signal if primary_fv else 0.0,
                "volume_ratio": predict_features["volume_ratio"],
                "bb_position": predict_features["bb_position"],
                "price_change_pct": predict_features["price_change_pct"],
            },
            "cascade": cascade, "macro": {}
        }

        votes = []
        for strate in [self.amd_strate, self.momentum_strate, self.structure_strate, self.macro_strate]:
            r = strate.safe_analyze(strate_data)
            if r.confidence > 0.3:
                votes.append(Vote(r.strate_name, r.direction, r.confidence, r.reasoning))

        if prediction.confidence > 0.3:
            votes.append(Vote("PREDICTIVE", prediction.direction, prediction.confidence,
                              f"Model: {prediction.model_version} p={prediction.probability:.2f}"))

        poly_vote = self.polymarket_strate.generate_parliament_vote(poly_opps, asset)
        if poly_vote.confidence > 0:
            votes.append(poly_vote)

        if asset == "BTC" and self.config.BTC_LATENCY_ARB_ENABLED:
            latency_signals = self.latency_arb.last_signals
            if latency_signals:
                lat_vote = self.latency_arb.generate_parliament_vote(latency_signals)
                if lat_vote.confidence > 0:
                    votes.append(lat_vote)

        if self.legacy_bridge.available_strates:
            try:
                for v in self.legacy_bridge.generate_votes(symbol, primary_ohlcv,
                                                            features=predict_features, trade_history=[]):
                    if v.confidence > 0.3:
                        votes.append(v)
            except Exception as e:
                logger.debug(f"Legacy bridge {symbol}: {e}")

        if not votes:
            return

        if self.narrator and primary_fv:
            price = primary_ohlcv[-1].close if primary_ohlcv else 0
            self.narrator.observe_market(symbol=symbol, price=price, rsi=primary_fv.rsi,
                                         trend=cascade.get("dominant_trend", "NEUTRAL"),
                                         volume_ratio=primary_fv.volume_ratio,
                                         orderbook_imbalance=sensory_input.orderbook_imbalance)

        decision = self.parliament.deliberate(votes)
        logger.info(f"{symbol}: {decision.direction} ({decision.strength:.0%}) | {len(votes)} votes")

        if self.narrator:
            hw = {s: self.hebbian.get_weight(s) for s in ["AMD", "MOMENTUM", "POLYMARKET", "LATENCY_ARB"]}
            await self.narrator.deliberate(symbol, votes, decision, hw)

        if decision.direction != "NEUTRAL":
            self.working_memory.push(decision.direction, decision.strength, "PARLIAMENT")

        self._update_active_signal(symbol, decision, self.config.PRIMARY_TIMEFRAME)

        if self.narrator:
            all_s = ["AMD", "MOMENTUM", "STRUCTURE", "MACRO", "POLYMARKET", "LATENCY_ARB",
                     "EPISTEMIQUE", "MINSKY", "REFLEXIVITE", "COMPORTEMENTAL", "FRACTAL"]
            await self.narrator.reflect([], {s: self.hebbian.get_weight(s) for s in all_s}, [])

        consensus = self.working_memory.get_consensus()
        if consensus and self.config.MODE in ["paper", "live"]:
            direction, confidence = consensus
            await self.execution.try_execute(
                symbol=symbol, direction=direction, confidence=confidence,
                decision=decision, feature_vector=primary_fv,
                binance=self.binance, narrator=self.narrator, alert_queue=self.alert_queue
            )

    async def _run_btc_latency_tick(self):
        try:
            signals = await self.latency_arb.scan()
            if signals:
                vote = self.latency_arb.generate_parliament_vote(signals)
                if vote.direction != "NEUTRAL" and vote.confidence > 0.3:
                    self.btc_working_memory.push(vote.direction, vote.confidence, "LATENCY_ARB")
                    if self.narrator:
                        await self.narrator.polymarket_signal(signals, vote)
                    if self.alert_queue and vote.confidence > 0.6:
                        top = signals[0]
                        self.alert_queue.alert_polymarket(
                            asset="BTC", direction=vote.direction, edge=top.edge,
                            confidence=top.confidence, question=top.question[:80],
                            kelly=top.kelly_fraction
                        )
        except Exception as e:
            logger.warning(f"BTC latency arb error: {e}")

    def _update_active_signal(self, symbol: str, decision, timeframe: str):
        self._active_signals = [s for s in self._active_signals if s["symbol"] != symbol]
        if decision.direction != "NEUTRAL" and decision.strength > 0.4:
            self._active_signals.append({
                "symbol": symbol, "direction": decision.direction,
                "confidence": decision.strength, "timeframe": timeframe,
                "source": "PARLIAMENT", "polymarket": decision.polymarket_alignment
            })

    # ── Interface publique ─────────────────────────────────────────────

    async def run(self):
        try:
            from oracle_v2.cycle_manager import CycleManager
        except ImportError:
            from cycle_manager import CycleManager

        await self._init_connectors()
        await self._init_telegram()

        if self.narrator:
            all_s = ["AMD", "MOMENTUM", "STRUCTURE", "MACRO", "POLYMARKET", "PREDICTIVE",
                     "LATENCY_ARB", "EPISTEMIQUE", "MINSKY", "REFLEXIVITE", "COMPORTEMENTAL", "FRACTAL"]
            self.narrator.session_start(self.config.MODE, self.config.TRADING_PAIRS,
                                        {s: self.hebbian.get_weight(s) for s in all_s})

        self._cycle_manager = CycleManager(self.config, self)
        await self._cycle_manager.run()
        await self._shutdown()

    async def _init_connectors(self):
        try:
            from oracle_v2.connectors.binance_connector import BinanceConnector
            from oracle_v2.connectors.capital_connector import CapitalConnector
        except ImportError:
            from connectors.binance_connector import BinanceConnector
            from connectors.capital_connector import CapitalConnector

        if self.config.BINANCE_API_KEY:
            self.binance = BinanceConnector(
                api_key=self.config.BINANCE_API_KEY, secret=self.config.BINANCE_SECRET,
                testnet=self.config.BINANCE_TESTNET
            )
            await self.binance.initialize()

        if self.config.CAPITAL_API_KEY:
            self.capital = CapitalConnector(
                api_key=self.config.CAPITAL_API_KEY, password=self.config.CAPITAL_PASSWORD,
                demo=(self.config.MODE != "live")
            )
            await self.capital.authenticate()

    async def _init_telegram(self):
        if not self.config.TELEGRAM_TOKEN:
            return
        try:
            from oracle_v2.telegram.alerts import AlertQueue
        except ImportError:
            from telegram.alerts import AlertQueue
        self.alert_queue = AlertQueue(token=self.config.TELEGRAM_TOKEN, chat_id=self.config.TELEGRAM_CHAT_ID)
        await self.alert_queue.start()

    async def _shutdown(self):
        logger.info("ORACLE v2 shutdown...")
        if self.binance:
            await self.binance.close()
        if self.capital:
            await self.capital.close()
        await self.polymarket_strate.close()
        await self.latency_arb.close()
        if self.alert_queue:
            await self.alert_queue.stop()

    def get_status(self) -> dict:
        return {
            "mode": self.config.MODE,
            "brainstem": self.brainstem.get_status_dict(),
            "open_positions": len(self.execution.open_positions),
            "active_signals": len(self._active_signals),
        }

    def get_active_signals(self) -> list:
        return self._active_signals.copy()

    def pause(self):
        if hasattr(self, "_cycle_manager"):
            self._cycle_manager.pause()

    def resume(self):
        if hasattr(self, "_cycle_manager"):
            self._cycle_manager.resume()
```
